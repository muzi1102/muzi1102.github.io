冒泡（js）

var a = [34, 203, 3, 746, 200, 984, 198, 764, 9];
function bubbleSort(a)
{
    var swapped;
    do {
        swapped = false;
        for (var i=0; i < a.length-1; i++) {
            if (a[i] > a[i+1]) {
                var temp = a[i];
                a[i] = a[i+1];
                a[i+1] = temp;
                swapped = true;
            }
        }
    } while (swapped);
}

bubbleSort(a);
console.log(a);

这是之前在谷歌的时候发现的方法。一看发现这个博客写了不少通俗易懂的算法。更多请点击http://www.stoimen.com/blog/
还有阮一峰的关于快排的介绍http://www.ruanyifeng.com/blog/2011/04/quicksort_in_javascript.html

数组排序

js按顺序排序（升序或降序）我一般使用如下方法

var arr = [1,2,99,3,4,59,6,71,8,9];
arr.sort(function(a,b){
    return a-b;
})
console.log(arr);

ps:这里是按升序排列。如果降序只需要把a-b改为b-a
更多的方法请点击：

  • http://www.ghugo.com/javascript-sort-array-objects/

获取所有孩子节点

平常我使用的是这种方法。当然。方法多种

function getChildEle(ele) {
      var childs=[];
      if (!(ele&&ele.nodeType&&ele.nodeType===1)) {
            return false;
       }
       for (var i = 0,len= ele.childNodes.length; i < len; i++) {
           if (ele.childNodes[i].nodeType===1) {
               childs.push(ele.childNodes[i])
           }
       }
       console.log(childs);

}

数组的去重

var arr=[2,3,4,1,1,5,6,6],
    arr1=[];
arr.map(function(item){
    if (arr1.indexOf(item)==-1) {
        arr1.push(item);
    }
})

ps:这种方法没有改变原来数组的值。
更多方法请点击http://jszen.com/best-way-to-get-unique-values-of-an-array-in-javascript.7.html

单例模式

首先看一段代码

  var Singleton=(function(){
      var instance;
      function createInstance() {
          var object = new Object("I am the instance");
        return object;
      }
      return {
          getInstance:function(){
              if (!instance) {
                  instance=createInstance();
              }
              return instance;
          }
      }
  })()
  function run() {

    var instance1 = Singleton.getInstance();
    var instance2 = Singleton.getInstance();

    alert("Same instance? " + (instance1 === instance2));  
}
run();

ps:(function(){})()这种方式能避免变量暴露。可能使用return{}暴露出来
更多请点击http://www.dofactory.com/javascript/singleton-design-pattern
